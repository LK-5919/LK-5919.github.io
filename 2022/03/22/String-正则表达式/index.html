String相关方法

字符串在底层中其实是以字符数组的形式保存的；
var str = "hello";
console.log(str.length);  //输出 5
console.log(str[1]);  //输出 e

			/*
			 * charAt()
			 * 	- 可以返回字符串中指定位置的字符
			 * 	- 根据索引获取指定的字符
             		该方法不太常用，需要使用索引获取单个字符的话，直接使用str[index]更加便捷；
			 */
var str = "helloworld";
console.log(str.charAt('3'));  //输出 l

			/*
			 * charCodeAt()
			 * 	- 获取指定位置字符的字符编码（Unicode编码）
			 */
var str = "helloLK"; 
console.log(str.charCodeAt(2));  //输出 108 即字符l的Unicode编码是108；

			/*
			 * String.formCharCode()
			 * 	- 可以根据字符编码去获取字符
			 */
console.log(String.fromCharCode(108));  //输出 字符 l；

			/*
			 * concat()
			 * 	- 可以用来连接两个或多个字符串
			 * 	- 作用和+一样
			 */
var str = "hello";
console.log(str.concat("world")); //输出 helloworld

			/*
			 * indexof()
			 * 	- 该方法可以检索一个字符串中是否含有指定内容
			 * 	- 如果字符串中含有该内容，则会返回其第一次出现的索引
			 * 		如果没有找到指定的内容，则返回-1
			 * 	- 可以指定一个第二个参数，指定开始查找的位置
			 * 
			 * lastIndexOf();
			 * 	- 该方法的用法和indexOf()一样，
			 * 		不同的是indexOf是从前往后找，
			 * 		而lastIndexOf是从后往前找  -- 返回的索引是从从左往右开始的
			 * 	- 也可以指定开始查找的位置
			 */
var str = "hello";
console.log(str.indexOf("a"));  //返回 -1 说明查找失败
console.log(str.indexOf("l"));  //返回 2 说明字符 l 在第三个字符
console.log(str.indexOf("l",3)); //返回3 说明从索引为3开始查找字符 l 
console.log(str.lastIndexOf("l"));  //返回 3 说明从后往前找 ，首次出现字符 l 在第4个字符

			/*
			 * slice()
			 * 	- 可以从字符串中截取指定的内容
			 * 	- 不会影响原字符串，而是将截取到内容返回
			 * 	- 参数：
			 * 		第一个，开始位置的索引（包括开始位置）
			 * 		第二个，结束位置的索引（不包括结束位置）
			 * 			- 如果省略第二个参数，则会截取到后边所有的
			 * 		- 也可以传递一个负数作为参数，负数的话将会从后边计算
			 */
var str = "helloworld";
var result = str.slice(1,2);
console.log(str);  //输出 helloworld
console.log(result);  // 输出 e
result = str.slice(2);
console.log(result); // 输出 lloworld
console.log(str.slice(2,-1)); //输出 lloworl  参数-1表示在倒数第一个位置停止截取

			/*
			 * substring()
			 * 	- 可以用来截取一个字符串，可以slice()类似
			 * 	- 参数：
			 * 		- 第一个：开始截取位置的索引（包括开始位置）
			 * 		- 第二个：结束位置的索引（不包括结束位置）
			 * 		- 不同的是这个方法不能接受负值作为参数，
			 * 			如果传递了一个负值，则默认使用0
			 * 		- 而且他还自动调整参数的位置，如果第二个参数小于第一个，则自动交换
			 */
var str = "helloworld";
console.log(str.substring(1,3)); //输出 el
console.log(str.substring(1,0)); // 输出 h 第二个参数小于第一个参数，自动调整为（0,1）

			/*
			 * substr()
			 * 	- 用来截取字符串
			 * 	- 参数：
			 * 		1.截取开始位置的索引
			 * 		2.截取的长度
			 */
var str = 'helloworld';
console.log(str.substr(1,3));  //输出 ell 表示从第二个字符开始截取是三个字符

			/*
			 * split()
			 * 	- 可以将一个字符串拆分为一个数组
			 * 	- 参数：
			 * 		-需要一个字符串作为参数，将会根据该字符串去拆分数组
			 */
var str = "helloworld";
var arr = str.split("w");
console.log(str); //输出 helloworld
console.log(Array.isArray(arr)); //输出 true
console.log(arr); // 输出 ['hello', 'orld']  从字符 w 处分开，且不包含字符 w 
console.log(str.split(""));  //输出  ['h', 'e', 'l', 'l', 'o', 'w', 'o', 'r', 'l', 'd']

			/*
			 * toUpperCase()
			 * 	- 将一个字符串转换为大写并返回
			 */
var str = "hedenLsdHGH";
console.log(str.toUpperCase());  //输出 HEDENLSDHGH

			/*
			 * toLowerCase()
			 * 	-将一个字符串转换为小写并返回
			 */
var str = "hedenLsdHGH";
console.log(str.toLowerCase());  //输出 hedenlsdhgh

正则表达式

//创建正则表达式对象 -- 使用构造函数创造
			/*
			 * 语法：
			 * 	var 变量 = new RegExp("正则表达式","匹配模式");
			 *  使用typeof检查正则对象，会返回object
			 * 	var reg = new RegExp("a"); 这个正则表达式可以来检查一个字符串中是否含有a
			 * 在构造函数中可以传递一个匹配模式作为第二个参数，
			 * 		可以是 
			 * 			i 忽略大小写 
			 * 			g 全局匹配模式
			 */
			/*
			 * 正则表达式的方法：
			 * 	test()
			 * 	 - 使用这个方法可以用来检查一个字符串是否符合正则表达式的规则，
			 * 		如果符合则返回true，否则返回false
			 */
var reg = new RegExp("ab","i");  //创建正则表达式规则 要求字符串中要包含有 "ab"，且不区分大小写
console.log(reg.test("sdxardbasds"));  //输出 false
console.log(reg.test("hsujabiijio"));  // 输出 true
console.log(reg.test("Absdjs"));  //输出 true

方括号

//使用字面量创建正则表达式
var reg = /ab/i;   //等同于new RegExp("ab","i");
console.log(reg.test("abhdjshuh")); //输出 true

			/*
				方括号
			 * 使用 | 表示或者的意思
			 * []里的内容也是或的关系
			 * [ab] == a|b
			 * [a-z] 任意小写字母
			 * [A-Z] 任意大写字母
			 * [A-z] 任意字母
			 * [0-9] 任意数字
			  * [^ ] 除了
			 */
reg = /a|b|c/;  //创建一个规则为包含字符 a 或 b 或 c 的正则对象
console.log(reg.test("asddefe"));  //输出 true 

reg = /[A-Z]/;  //创建一个规则为包含字符 A 到 Z 的正则对象
console.log(reg.test("asjjHsBd"));  //返回 true

reg = /[^BS]/;  //创建一个规则为除了字符BS以外的正则对象
console.log(reg.test("Bdfgs"));  //输出 true 因为该字符串中BS外还有其他字符
console.log(reg.test("AHYGdfsb")); //输出 true
console.log(reg.test("B"));  // 输出false

元字符

			/*
			 * \w
			 * 	- 任意字母、数字、_  [A-z0-9_]
			 * \W
			 * 	- 除了字母、数字、_  [^A-z0-9_]
			 * \d
			 * 	- 任意的数字 [0-9]
			 * \D
			 * 	- 除了数字 [^0-9]
			 * \s
			 * 	- 空格
			 * \S
			 * 	- 除了空格
			 * \b
			 * 	- 单词边界
			 * \B
			 * 	- 除了单词边界
			 */
var reg = /\w/;  //创建一个规则为包含任意字母，数字，_的正则对象；
console.log(reg.test("!###@"));  //输出 false
console.log(reg.test("#$$%asdf432"));  //输出 true

var reg = /\s/; //创建一个规则为包含空格的正则对象；
console.log(reg.test("hsdjhsch"));   //输出 false
console.log(reg.test("fgdsjk jkjlko"));  //输出 true

var reg = /\b\w{3}\b/;  //创建一个包含单词边界的正则对象；
console.log(reg.test("sdfdsf")); //输出 false
console.log(reg.test(" jhf jsd h fkgjksfdjfio"));  //输出 true

量词

/*
			 * 量词
			 * 	- 通过量词可以设置一个内容出现的次数
			 * 	- 量词只对它前边的一个内容起作用
			 * 	- {n} 正好出现n次
			 * 	- {m,n} 出现m-n次
			 * 	- {m,} m次以上
			 * 	- + 至少一个，相当于{1,}
			 * 	- * 0个或多个，相当于{0,}
			 * 	- ? 0个或1个，相当于{0,1}
			 */
var reg = /[ab]{3}/;  //创建一个规则为连续出现三次字符a或者字符b的正则对象
console.log(reg.test("adsd"));  //输出 false
console.log(reg.test("aabbbsd"));  //输出 true

var reg = /s{1,3}/;  //创建一个规则为连续出现1到三次的字符s的正则对象
console.log(reg.test("fgfdhg"));  //输出 false
console.log(reg.tets("dddssgf")); // 输出 true
console.log(reg.test("ssss"));  //输出 true  -- 当虽然这里有4个s字符，但是在未开启全局模式时
							//还没开始检测第四个字符就会返回true
							//如果要求严格控制在1到3次，需要开启全局匹配模式

			/*
			 * 	^ 表示开头
			 * 	$ 表示结尾
			 */
var reg = /^a|b$/;  //创建一个规则为以a字符开头或者b字符结尾的正则对象
console.log(reg.test("dfdffsf"));  //输出 false
console.log(reg.test("adsdjjd"));  //输出 true
console.log(reg.test("dsfsb"));   //输出 true

RegExp方法

			/*
			 * split()
			 * 	- 可以将一个字符串拆分为一个数组
			 * 	- 方法中可以传递一个正则表达式作为参数，这样方法将会根据正则表达式去拆分字符串
			 * 	- 这个方法即使不指定全局匹配，也会全都插分
			 */
var str = "helloworld";
var result = str.split(/[lr]/);  //根据字母lr来拆分字符串；
console.log(result);  //输出  ['he', '', 'owo', '', 'd']
console.log(result instanceof Array); //输出true

			/*
			 * search()
			 * 	- 可以搜索字符串中是否含有指定内容
			 * 	- 如果搜索到指定内容，则会返回第一次出现的索引，如果没有搜索到返回-1
			 * 	- 它可以接受一个正则表达式作为参数，然后会根据正则表达式去检索字符串
			 * 	- serach()只会查找第一个，即使设置全局匹配也没用
			 */
var str = "heooljsdhfuhdsftFFDDDhnduhasui";
console.log(str.search(/f[adHt]f/i));  //输出 14

			/*
			 * match()
			 * 	- 可以根据正则表达式，从一个字符串中将符合条件的内容提取出来
			 * 	- 默认情况下我们的match只会找到第一个符合要求的内容，找到以后就停止检索
			 * 		我们可以设置正则表达式为全局匹配模式，这样就会匹配到所有的内容
			 * 		可以为一个正则表达式设置多个匹配模式，且顺序无所谓
			 * 	- match()会将匹配到的内容封装到一个数组中返回，即使只查询到一个结果
			 */
var str = "heooljsdhfuhdsftFFDDDhnduhasui";
console.log(str.match(/f[adHt]f/i));//输出 ['ftF', index: 14, input:'heooljsdhfuhdsftFFDDDhnduhasui']

			/*
			 * replace()
			 * 	- 可以将字符串中指定内容替换为新的内容
			 *  - 参数：
			 * 		1.被替换的内容，可以接受一个正则表达式作为参数
			 * 		2.新的内容
			 *  - 默认只会替换第一个
			 */
var str = "heooljsdhfuhdsftFFDDDhnduhasui";
console.log(str.replace(/f[adHt]f/gi,"@")); //输出 heooljsdhfuhds@FDDDhnduhasui
