vim /etc/init.d/nginx

#脚本内容 如下：
#!/bin/sh
#
# nginx - this script starts and stops the nginx daemon
#
# chkconfig:   - 85 15
# description:  NGINX is an HTTP(S) server, HTTP(S) reverse \
#               proxy and IMAP/POP3 proxy server
# processname: nginx
# config:      /etc/nginx/nginx.conf
# config:      /etc/sysconfig/nginx
# pidfile:     /var/run/nginx.pid

# Source function library.
. /etc/rc.d/init.d/functions

# Source networking configuration.
. /etc/sysconfig/network

# Check that networking is up.
[ "$NETWORKING" = "no" ] && exit 0

nginx="/usr/sbin/nginx"
prog=$(basename $nginx)

NGINX_CONF_FILE="/etc/nginx/nginx.conf"

[ -f /etc/sysconfig/nginx ] && . /etc/sysconfig/nginx

lockfile=/var/lock/subsys/nginx

make_dirs() {
   # make required directories
   user=`$nginx -V 2>&1 | grep "configure arguments:.*--user=" | sed 's/[^*]*--user=\([^ ]*\).*/\1/g' -`
   if [ -n "$user" ]; then
      if [ -z "`grep $user /etc/passwd`" ]; then
         useradd -M -s /bin/nologin $user
      fi
      options=`$nginx -V 2>&1 | grep 'configure arguments:'`
      for opt in $options; do
          if [ `echo $opt | grep '.*-temp-path'` ]; then
              value=`echo $opt | cut -d "=" -f 2`
              if [ ! -d "$value" ]; then
                  # echo "creating" $value
                  mkdir -p $value && chown -R $user $value
              fi
          fi
       done
    fi
}

start() {
    [ -x $nginx ] || exit 5
    [ -f $NGINX_CONF_FILE ] || exit 6
    make_dirs
    echo -n $"Starting $prog: "
    daemon $nginx -c $NGINX_CONF_FILE
    retval=$?
    echo
    [ $retval -eq 0 ] && touch $lockfile
    return $retval
}

stop() {
    echo -n $"Stopping $prog: "
    killproc $prog -QUIT
    retval=$?
    echo
    [ $retval -eq 0 ] && rm -f $lockfile
    return $retval
}

restart() {
    configtest || return $?
    stop
    sleep 1
    start
}

reload() {
    configtest || return $?
    echo -n $"Reloading $prog: "
    killproc $prog -HUP
    retval=$?
    echo
}

force_reload() {
    restart
}

configtest() {
  $nginx -t -c $NGINX_CONF_FILE
}

rh_status() {
    status $prog
}

rh_status_q() {
    rh_status >/dev/null 2>&1
}

case "$1" in
    start)
        rh_status_q && exit 0
        $1
        ;;
    stop)
        rh_status_q || exit 0
        $1
        ;;
    restart|configtest)
        $1
        ;;
    reload)
        rh_status_q || exit 7
        $1
        ;;
    force-reload)
        force_reload
        ;;
    status)
        rh_status
        ;;
    condrestart|try-restart)
        rh_status_q || exit 0
            ;;
    *)
        echo $"Usage: $0 {start|stop|status|restart|condrestart|try-restart|reload|force-reload|configtest}"
        exit 2
esac

#进入vim编辑器后，按i进入插入模式，再按shift+insert键粘贴脚本代码

#修改其中的代码为如下
nginx="/usr/nginx/sbin/nginx"  #自己的路径
NGINX_CONF_FILE="/usr/nginx/conf/nginx.conf"  #自己的路径

#保存并退出文件后设置文件执行权限

Nginx跨域问题

Vue项目存在两种跨域问题：开发环境下的跨域问题和线上环境下的跨域问题

Vue开发环境跨域问题

修改原来的前端项目文件，在项目根目录下新增文件 vue.config.js ，具体配置如下

添加如下代码：

module.exports = {
    //关闭eslint
    lintOnSave:false,
    //配置代理跨域
  devServer: {
    proxy: {
      "/api": {
        target: "http://175.178.92.227:8082/",
      },
    },
  },
}

之后，vue项目中的axios请求就会发往代理IP地址的。

Vue线上环境的跨域问题

通过使用nginx代理解决跨域问题，在vue项目中axios发送请求的IP和端口应该和打包后访问dist资源的端口一致。

以本项目为例，我将打包后的dist放在服务器的 /web/onlineExam-Front/ 下，给nginx的配置如下：

#onlineExam-Front
	server {
           listen 8081;
           server_name localhost 127.0.0.1;  #服务器IP
	       #允许跨域请求的域，* 代表所有
     	   #add_header 'Access-Control-Allow-Origin' *;
     	   #允许带上cookie请求
     	   #add_header 'Access-Control-Allow-Credentials' 'true';
           #允许请求的方法，比如 GET/POST/PUT/DELETE
           #add_header 'Access-Control-Allow-Methods' *;
           #允许请求的header
           #add_header 'Access-Control-Allow-Headers' *;
           #root /web/onlineExam-Front/dist; #vue项目路径
           #index index.html;
	   
          location / {
          	 add_header 'Access-Control-Allow-Origin' '$http_origin' ;
          	 add_header 'Access-Control-Allow-Credentials' 'true' ;
          	 add_header 'Access-Control-Allow-Methods' 'PUT,POST,GET,DELETE,OPTIONS' ;
          	 add_header 'Access-Control-Allow-Headers' 'Content-Type,Content-Length,Authorization,Accept,X-Requested-With';

            root /web/onlineExam-Front/dist; #vue项目路径
            index index.html;

             # 此处用于处理 Vue、Angular、React 使用H5 的 History时 重写的问题
            # if (!-e $request_filename) {
            #     rewrite ^(.*) /index.html last;
            #     break;
            # }
	       }

这样，当在浏览器通过IP+8081端口访问时，将会向用户展示dist文件夹下的index.html。

而我们在上线时，应该保持axios的请求路径也是同一个IP+端口，具体如下

//axios请求基本路径和dist资源访问路径应保持一致
const baseURL = 'http://175.178.92.227:8081/api/';

const service = axios.create({
  baseURL: baseURL,
  timeout: 5000
})

之后再通过nginx给刚才的dist再添加一个location配置，如下

# 代理服务端接口
        location /api/ {
            proxy_pass http://localhost:8082/api/;# 代理接口地址
        }
        
#此时nginx.conf 全部文件如下

worker_processes  1;

events {
    worker_connections  1024;
}


http {
    include       mime.types;
    default_type  application/octet-stream;
    sendfile        on;
    #tcp_nopush     on;
    #keepalive_timeout  0;
    keepalive_timeout  65;
    #gzip  on;
    
    #nginx默认首页
    server {
        listen       8888;
        server_name  localhost;
        
        location / {
            root   html;
            index  index.html index.htm;
        }
        error_page   500 502 503 504  /50x.html;
        location = /50x.html {
            root   html;
        }
    }
    
	#onlineExam
	server {
           listen 8081;
           server_name localhost 127.0.0.1;  #服务器IP
	   
          location / {
          	 add_header 'Access-Control-Allow-Origin' '$http_origin' ;
          	 add_header 'Access-Control-Allow-Credentials' 'true' ;
          	 add_header 'Access-Control-Allow-Methods' 'PUT,POST,GET,DELETE,OPTIONS' ;
          	 add_header 'Access-Control-Allow-Headers' 'Content-Type,Content-Length,Authorization,Accept,X-Requested-With';

            root /web/onlineExam-Front/dist; #vue项目路径
            index index.html;
	       }

	    #解决跨域 -- 注意前端项目的请求路径应该是8081端口，先提交到8081端口，代理服务器通过/api/ 前缀
	    #识别请求，将请求代理转发到服务器上的8082端口，从而解决跨域
	    
        # 代理服务端接口--将转发到本服务器上运行的SpringBoot项目端口
        location /api/ {
            proxy_pass http://localhost:8082/api/;# 代理接口地址
            }
         }
}

这样配置以后，在访问dist下的网页时，在发送具有api请求前缀的请求时，就会被nginx代理接口识别，并转发到此时服务器的8082/api/下的端口（8082/api/ 为此时项目SpringBoot后端在服务器运行的端口路径），从而解决跨域问题。
